// main.js - Pablo Talavera Portfolio
// Core functionality and utilities

'use strict';

console.log('Portfolio initialized - v2.0 with Testimonials, Booking & Media Kit');

// Booking data retrieval
function getAllBookings() {
    const bookings = [];
    for (let i = 0; i < localStorage.length; i++) {
        const key = localStorage.key(i);
        if (key.startsWith('booking_')) {
            try {
                bookings.push(JSON.parse(localStorage.getItem(key)));
            } catch (e) {
                console.error('Error parsing booking:', e);
            }
        }
    }
    return bookings.sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp));
}

// Admin export function
function exportBookingData() {
    const bookings = getAllBookings();
    const dataStr = JSON.stringify(bookings, null, 2);
    const dataBlob = new Blob([dataStr], { type: 'application/json' });
    const url = URL.createObjectURL(dataBlob);
    const link = document.createElement('a');
    link.href = url;
    link.download = 'booking-data-' + new Date().toISOString().split('T')[0] + '.json';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
}

// Page scroll tracking
document.addEventListener('DOMContentLoaded', () => {
    console.log('✓ DOM loaded');
    console.log('✓ Total bookings:', getAllBookings().length);
});

// Performance monitoring
if (window.performance) {
    window.addEventListener('load', () => {
        const perfData = window.performance.timing;
        const pageLoadTime = perfData.loadEventEnd - perfData.navigationStart;
        console.log('✓ Page load time:', pageLoadTime, 'ms');
    });
}

// Utility: Format date
function formatDate(dateString) {
    return new Date(dateString).toLocaleDateString('en-US', {
        year: 'numeric',
        month: 'long',
        day: 'numeric'
    });
}

// Utility: Validate email
function validateEmail(email) {
    const re = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return re.test(email);
}

// Export for use in console
window.portfolioUtils = {
    getAllBookings,
    exportBookingData,
    formatDate,
    validateEmail
};
